簡單寫一個批次檔來測試

@echo off
set MSBUILD="C:\Program Files (x86)\MSBuild\12.0\bin\amd64\msbuild.exe"
if not exist %MSBUILD% (
 echo MSBuild not found
 exit
) 

%MSBUILD% WebApplication1.sln /p:Configuration=Release /p:RunOctoPack=true /m
pause
